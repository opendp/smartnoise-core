pub fn log_spaced(length: usize, start_exp: f64, stop_exp: f64) -> Vec<f64>

Generates a base 10 log spaced vector of the given length between the specified decade exponents (inclusive). Equivalent to MATLAB logspace

Examples

use statrs::generate;

let x = generate::log_spaced(5, 0.0, 4.0);
assert_eq!(x, [1.0, 10.0, 100.0, 1000.0, 10000.0]);
