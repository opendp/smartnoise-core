pub trait Assign<Src = Self> {
    fn assign(&mut self, src: Src);
}

Assigns to a number from another value.

Examples

use rug::Assign;
struct I(i32);
impl Assign<i16> for I {
    fn assign(&mut self, rhs: i16) {
        self.0 = rhs as i32;
    }
}
let mut i = I(0);
i.assign(42_i16);
assert_eq!(i.0, 42);
