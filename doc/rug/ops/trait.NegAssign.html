pub trait NegAssign {
    fn neg_assign(&mut self);
}

Compound negation and assignment.

Examples

use rug::ops::NegAssign;
struct I(i32);
impl NegAssign for I {
    fn neg_assign(&mut self) {
        self.0 = -self.0;
    }
}
let mut i = I(42);
i.neg_assign();
assert_eq!(i.0, -42);
